## 4. tldraw 核心集成

### 4.1 tldraw 简介

tldraw 是一个功能强大的开源白板引擎，提供：

| 功能 | 说明 |
|------|------|
| 形状系统 | 内置多种形状，支持自定义扩展 |
| 工具系统 | 选择、绘图、文字等工具，支持自定义 |
| 事件系统 | 完整的交互事件处理 |
| 撤销/重做 | 内置历史记录管理 |
| 协作支持 | 支持多人实时协作（可选） |
| 导出功能 | 支持导出图片、JSON等 |

### 4.2 自定义形状开发

所有自定义形状继承 tldraw 的 `ShapeUtil` 类：

```typescript
// src/components/shapes/base/CustomShape.ts

import { ShapeUtil, TLBaseShape, HTMLContainer } from 'tldraw'

// 定义形状类型
export type MyCustomShapeProps = {
  w: number
  h: number
  rotation: number
  fill: string
  stroke: string
  strokeWidth: number
}

export const MyCustomShape = createShapeId('my-custom')

export class MyCustomShapeUtil extends ShapeUtil<MyCustomShape> {
  // 形状配置
  static type = MyCustomShape
  static props: ShapeProps<MyCustomShapeProps> = {
    w: ShapeProps.number,
    h: ShapeProps.number,
    rotation: ShapeProps.number,
    fill: ShapeProps.string,
    stroke: ShapeProps.string,
    strokeWidth: ShapeProps.number,
  }

  // 默认属性
  getDefaultProps(): MyCustomShapeProps {
    return {
      w: 100,
      h: 100,
      rotation: 0,
      fill: '#ffffff',
      stroke: '#000000',
      strokeWidth: 2,
    }
  }

  // 渲染组件
  component(shape: MyCustomShape) {
    return (
      <HTMLContainer>
        <div
          style={{
            width: shape.props.w,
            height: shape.props.h,
            backgroundColor: shape.props.fill,
            border: `${shape.props.strokeWidth}px solid ${shape.props.stroke}`,
          }}
        />
      </HTMLContainer>
    )
  }

  // 碰撞检测
  getBounds(shape: MyCustomShape): Box {
    return new Box(0, 0, shape.props.w, shape.props.h)
  }

  // 绑定点（用于连接线）
  getBindings(shape: MyCustomShape): Binding[] {
    return [
      { id: 'top', x: shape.props.w / 2, y: 0 },
      { id: 'bottom', x: shape.props.w / 2, y: shape.props.h },
      { id: 'left', x: 0, y: shape.props.h / 2 },
      { id: 'right', x: shape.props.w, y: shape.props.h / 2 },
    ]
  }
}
```

### 5.1 图形分割模块

使用 PolyBool.js 实现图形分割功能：

```typescript
// src/utils/polygon.ts

import PolyBool from 'polybooljs'
import { Box, Point } from 'tldraw'

export interface Polygon {
  regions: Point[][]
  inverted: boolean
}

export class PolygonUtils {
  /**
   * 多边形分割
   * @param source 源多边形
   * @param line 分割线
   * @returns 分割后的多边形数组
   */
  static split(source: Point[], line: Point[]): Point[][] {
    // 将源多边形转换为 PolyBool 格式
    const sourcePoly = this.toPolyBool(source)
    
    // 创建分割线形成的切割多边形
    const cutPoly = this.createCutPolygon(line)
    
    // 计算分割结果
    const result1 = PolyBool.intersect(sourcePoly, cutPoly)
    const result2 = PolyBool.difference(sourcePoly, cutPoly)
    
    return [
      this.fromPolyBool(result1),
      this.fromPolyBool(result2),
    ].filter(p => p.length >= 3)
  }

  /**
   * 多边形合并
   */
  static union(polygons: Point[][]): Point[][] {
    const result = polygons.reduce((acc, poly) => {
      return PolyBool.union(acc, this.toPolyBool(poly))
    }, { regions: [], inverted: false })
    
    return this.fromPolyBoolMultiple(result)
  }

  /**
   * 多边形差集
   */
  static difference(source: Point[], clip: Point[]): Point[][] {
    const result = PolyBool.difference(
      this.toPolyBool(source),
      this.toPolyBool(clip)
    )
    return this.fromPolyBoolMultiple(result)
  }

  /**
   * 多边形交集
   */
  static intersect(source: Point[], clip: Point[]): Point[][] {
    const result = PolyBool.intersect(
      this.toPolyBool(source),
      this.toPolyBool(clip)
    )
    return this.fromPolyBoolMultiple(result)
  }

  // 转换函数
  private static toPolyBool(points: Point[]): Polygon {
    return {
      regions: [points.map(p => [p.x, p.y])],
      inverted: false,
    }
  }

  private static fromPolyBool(poly: Polygon): Point[] {
    if (!poly.regions || poly.regions.length === 0) return []
    return poly.regions[0].map(([x, y]) => ({ x, y }))
  }

  private static fromPolyBoolMultiple(poly: Polygon): Point[][] {
    if (!poly.regions) return []
    return poly.regions.map(region => 
      region.map(([x, y]) => ({ x, y }))
    )
  }

  private static createCutPolygon(line: Point[]): Polygon {
    // 创建一个足够大的矩形来切割
    const extend = 10000
    const [p1, p2] = line
    const dx = p2.x - p1.x
    const dy = p2.y - p1.y
    const len = Math.sqrt(dx * dx + dy * dy)
    const nx = -dy / len * extend
    const ny = dx / len * extend

    return {
      regions: [[
        [p1.x + nx, p1.y + ny],
        [p2.x + nx, p2.y + ny],
        [p2.x - nx, p2.y - ny],
        [p1.x - nx, p1.y - ny],
      ]],
      inverted: false,
    }
  }
}
```

### 5.2 旋转模块

```typescript
// src/utils/transform.ts

import { Point, Shape } from 'tldraw'

export class TransformUtils {
  /**
   * 围绕指定点旋转
   * @param point 要旋转的点
   * @param center 旋转中心
   * @param angle 旋转角度（弧度）
   */
  static rotateAroundPoint(point: Point, center: Point, angle: number): Point {
    const cos = Math.cos(angle)
    const sin = Math.sin(angle)
    const dx = point.x - center.x
    const dy = point.y - center.y
    
    return {
      x: center.x + dx * cos - dy * sin,
      y: center.y + dx * sin + dy * cos,
    }
  }

  /**
   * 批量旋转点集
   */
  static rotatePoints(points: Point[], center: Point, angle: number): Point[] {
    return points.map(p => this.rotateAroundPoint(p, center, angle))
  }

  /**
   * 获取形状的旋转中心点
   */
  static getRotationCenter(shape: Shape): Point {
    const bounds = shape.getBounds()
    return {
      x: bounds.x + bounds.w / 2,
      y: bounds.y + bounds.h / 2,
    }
  }

  /**
   * 设置自定义旋转中心
   */
  static setCustomRotationCenter(shape: Shape, center: Point): void {
    shape.meta.rotationCenter = center
  }
}
```

```typescript
export interface ConnectionPoint {
  id: string
  position: Point
  direction: 'input' | 'output' | 'both'
}

export abstract class ConnectableShapeUtil extends ShapeUtil {
  // 连接点配置
  protected connectionPoints: ConnectionPoint[] = []

  /**
   * 获取所有连接点
   */
  getConnectionPoints(shape: Shape): Point[] {
    return this.connectionPoints.map(cp => ({
      x: shape.x + cp.position.x,
      y: shape.y + cp.position.y,
    }))
  }

  /**
   * 获取最近的连接点
   */
  getNearestConnectionPoint(shape: Shape, point: Point): ConnectionPoint | null {
    let nearest: ConnectionPoint | null = null
    let minDistance = Infinity

    for (const cp of this.connectionPoints) {
      const cpPosition = {
        x: shape.x + cp.position.x,
        y: shape.y + cp.position.y,
      }
      const distance = this.getDistance(cpPosition, point)
      
      if (distance < minDistance) {
        minDistance = distance
        nearest = cp
      }
    }

    return nearest
  }

  /**
   * 渲染连接点指示器
   */
  renderConnectionPoints(shape: Shape): JSX.Element {
    return (
      <g class="connection-points">
        {this.connectionPoints.map(cp => (
          <circle
            cx={shape.x + cp.position.x}
            cy={shape.y + cp.position.y}
            r={5}
            fill="#4a90d9"
            stroke="#ffffff"
            strokeWidth={2}
            class="connection-point"
          />
        ))}
      </g>
    )
  }

  private getDistance(p1: Point, p2: Point): number {
    return Math.sqrt((p1.x - p2.x) ** 2 + (p1.y - p2.y) ** 2)
  }
```

```typescript
// src/utils/snap.ts

import { Shape, Point } from 'tldraw'

export interface SnapResult {
  point: Point
  snapped: boolean
  guides: SnapGuide[]
}

export interface SnapGuide {
  type: 'horizontal' | 'vertical' | 'distance'
  start: Point
  end: Point
  value: number
}

export class SnapUtils {
  private snapDistance: number = 10 // 吸附距离阈值

  /**
   * 计算吸附点
   */
  calculateSnap(
    movingShape: Shape,
    targetPoint: Point,
    allShapes: Shape[]
  ): SnapResult {
    const guides: SnapGuide[] = []
    let snappedPoint = { ...targetPoint }
    let snapped = false

    // 收集所有参考点
    const referencePoints = this.collectReferencePoints(allShapes, movingShape.id)

    // 水平吸附
    const horizontalSnap = this.findHorizontalSnap(targetPoint, referencePoints)
    if (horizontalSnap) {
      snappedPoint.y = horizontalSnap.value
      guides.push(horizontalSnap.guide)
      snapped = true
    }

    // 垂直吸附
    const verticalSnap = this.findVerticalSnap(targetPoint, referencePoints)
    if (verticalSnap) {
      snappedPoint.x = verticalSnap.value
      guides.push(verticalSnap.guide)
      snapped = true
    }

    // 等距吸附
    const distanceSnap = this.findDistanceSnap(movingShape, allShapes)
    if (distanceSnap) {
      guides.push(...distanceSnap)
      snapped = true
    }

    return { point: snappedPoint, snapped, guides }
  }

  /**
   * 收集参考点
   */
  private collectReferencePoints(shapes: Shape[], excludeId: string): Point[] {
    const points: Point[] = []
    
    for (const shape of shapes) {
      if (shape.id === excludeId) continue
      
      const bounds = shape.getBounds()
      // 添加边界点：四个角 + 四个边中点
      points.push(
        { x: bounds.x, y: bounds.y }, // 左上
        { x: bounds.x + bounds.w, y: bounds.y }, // 右上
        { x: bounds.x, y: bounds.y + bounds.h }, // 左下
        { x: bounds.x + bounds.w, y: bounds.y + bounds.h }, // 右下
        { x: bounds.x + bounds.w / 2, y: bounds.y }, // 上中
        { x: bounds.x + bounds.w / 2, y: bounds.y + bounds.h }, // 下中
        { x: bounds.x, y: bounds.y + bounds.h / 2 }, // 左中
        { x: bounds.x + bounds.w, y: bounds.y + bounds.h / 2 }, // 右中
      )
    }
    
    return points
  }
```

```typescript
            stroke="#000"
            strokeWidth={1}
            fill="none"
          />
        )}
      </svg>
    )
  }

      path += ` L ${points[i].x} ${points[i].y}`

  }
```

```typescript
// src/types/animation.types.ts

export interface Animation {

  properties: {

    y?: number

    scaleX?: number
    scaleY?: number
    opacity?: number
    color?: string
  }
  easing?: EasingFunction
}

export type AnimationType =

  | 'rotate'      // 旋转
  | 'scale'       // 缩放
  | 'fade'        // 淡入淡出
  | 'path'        // 路径动画
  | 'custom'      // 自定义

// src/services/animation.service.ts

import { TldrawEditor, Shape } from 'tldraw'

export class AnimationService {
  private editor: TldrawEditor
  private animations: Map<string, Animation> = new Map()
  private isPlaying: boolean = false
  private currentTime: number = 0
  private animationFrame: number | null = null

  constructor(editor: TldrawEditor) {
    this.editor = editor
  }

  /**
   * 创建动画
   */
  createAnimation(config: Omit<Animation, 'id'>): Animation {
    const animation: Animation = {
      id: uuidv4(),
      ...config,
    }
    this.animations.set(animation.id, animation)
    return animation
  }

  /**
   * 播放动画
   */
  play(animationId?: string) {
    if (animationId) {
      this.playSingle(animationId)
    } else {
      this.playAll()
    }
  }

  /**
   * 播放单个动画
   */
  private playSingle(animationId: string) {
    const animation = this.animations.get(animationId)
    if (!animation) return

    const startTime = performance.now()
    
    const animate = (currentTime: number) => {
      const elapsed = currentTime - startTime
      const progress = Math.min(elapsed / animation.duration, 1)
      
      // 更新形状属性
      this.applyKeyframes(animation, progress)
      
      if (progress < 1) {
        this.animationFrame = requestAnimationFrame(animate)
      }
    }
    
    this.animationFrame = requestAnimationFrame(animate)
  }

  /**
   * 应用关键帧
   */
  private applyKeyframes(animation: Animation, progress: number) {
    const { keyframes, targets } = animation
    
    // 找到当前进度的前后关键帧
    const prevKeyframe = keyframes.find((kf, i) => {
      const next = keyframes[i + 1]
      return kf.time <= progress && (!next || next.time > progress)
    })
    
    const nextKeyframe = keyframes.find(kf => kf.time >= progress)
    
    if (!prevKeyframe || !nextKeyframe) return
    
    // 计算插值
    const localProgress = (progress - prevKeyframe.time) / (nextKeyframe.time - prevKeyframe.time)
    const easedProgress = this.applyEasing(localProgress, prevKeyframe.easing)
    
    // 应用到目标形状
    for (const shapeId of targets) {
      const shape = this.editor.getShape(shapeId)
      if (!shape) continue
      
      const interpolatedProps = this.interpolate(
        prevKeyframe.properties,
        nextKeyframe.properties,
        easedProgress
      )
      
      this.editor.updateShape({
        id: shapeId,
        type: shape.type,
        props: interpolatedProps,
      })
    }
  }

  /**
   * 插值计算
   */
  private interpolate(
    start: Record<string, any>,
    end: Record<string, any>,
    progress: number
  ): Record<string, any> {
    const result: Record<string, any> = {}
    
    for (const key in start) {
      if (key in end) {
        result[key] = start[key] + (end[key] - start[key]) * progress
      }
    }
    
    return result
  }

  /**
   * 应用缓动函数
   */
  private applyEasing(progress: number, easing?: string): number {
    switch (easing) {
      case 'ease':
        return this.easeInOut(progress)
      case 'ease-in':
        return this.easeIn(progress)
      case 'ease-out':
        return this.easeOut(progress)
      case 'linear':
      default:
        return progress
    }
  }

  private easeIn(t: number): number {
    return t * t
  }

  private easeOut(t: number): number {
    return t * (2 - t)
  }

  private easeInOut(t: number): number {
    return t < 0.5 ? 2 * t * t : -1 + (4 - 2 * t) * t
  }

  /**
   * 暂停动画
   */
  pause() {
    if (this.animationFrame) {
      cancelAnimationFrame(this.animationFrame)
      this.animationFrame = null
    }
    this.isPlaying = false
  }

  /**
   * 停止动画
   */
  stop() {
    this.pause()
    this.currentTime = 0
  }

  /**
   * 设置播放进度
   */
  setProgress(progress: number) {
    this.currentTime = progress
    // 应用当前进度的动画状态
    this.animations.forEach(animation => {
      this.applyKeyframes(animation, progress)
    })
  }
}
```

```typescript
// src/services/sync.service.ts

import { TldrawEditor, Record } from 'tldraw'

export interface SyncConfig {
  serverUrl: string
  userId: string
  notebookId: string
}

export interface SyncStatus {
  lastSyncTime: number
  pendingChanges: number
  isSyncing: boolean
  error?: string
}

export class SyncService {
  private config: SyncConfig
  private editor: TldrawEditor
  private status: SyncStatus = {
    lastSyncTime: 0,
    pendingChanges: 0,
    isSyncing: false,
  }
  private syncInterval: number | null = null

  constructor(editor: TldrawEditor, config: SyncConfig) {
    this.editor = editor
    this.config = config
    this.setupListeners()
  }

  /**
   * 设置监听器
   */
  private setupListeners() {
    // 监听本地变更
    this.editor.store.listen((changes) => {
      this.onLocalChange(changes)
    })
  }

  /**
   * 本地变更处理
   */
  private onLocalChange(changes: { added: Record[]; updated: Record[]; removed: string[] }) {
    this.status.pendingChanges++
    
    // 存储到本地队列
    this.storePendingChanges(changes)
  }

  /**
   * 开始同步
   */
  async startSync(intervalMs: number = 30000) {
    // 立即同步一次
    await this.sync()
    
    // 设置定时同步
    this.syncInterval = setInterval(() => {
      this.sync()
    }, intervalMs) as unknown as number
  }

  /**
   * 停止同步
   */
  stopSync() {
    if (this.syncInterval) {
      clearInterval(this.syncInterval)
      this.syncInterval = null
    }
  }

  /**
   * 执行同步
   */
  async sync(): Promise<void> {
    if (this.status.isSyncing) return
    
    this.status.isSyncing = true
    
    try {
      // 获取本地待同步变更
      const pendingChanges = await this.getPendingChanges()
      
      if (pendingChanges.length === 0) {
        // 拉取远程变更
        await this.pullChanges()
      } else {
        // 推送本地变更
        await this.pushChanges(pendingChanges)
      }
      
      this.status.lastSyncTime = Date.now()
      this.status.pendingChanges = 0
      this.status.error = undefined
    } catch (error) {
      this.status.error = error.message
      throw error
    } finally {
      this.status.isSyncing = false
    }
  }

  /**
   * 推送变更到服务器
   */
  private async pushChanges(changes: any[]): Promise<void> {
    const response = await fetch(`${this.config.serverUrl}/sync/push`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${await this.getAuthToken()}`,
      },
      body: JSON.stringify({
        notebookId: this.config.notebookId,
        changes,
        timestamp: Date.now(),
      }),
    })
    
    if (!response.ok) {
      throw new Error('Push failed')
    }
  }

  /**
   * 拉取服务器变更
   */
  private async pullChanges(): Promise<void> {
    const response = await fetch(
      `${this.config.serverUrl}/sync/pull?notebookId=${this.config.notebookId}&since=${this.status.lastSyncTime}`,
      {
        headers: {
          'Authorization': `Bearer ${await this.getAuthToken()}`,
        },
      }
    )
    
    if (!response.ok) {
      throw new Error('Pull failed')
    }
    
    const data = await response.json()
    
    // 应用远程变更到本地
    this.applyRemoteChanges(data.changes)
  }

  /**
   * 应用远程变更
   */
  private applyRemoteChanges(changes: any[]) {
    for (const change of changes) {
      this.editor.store.apply(change)
    }
  }

  /**
   * 存储待同步变更
   */
  private async storePendingChanges(changes: any) {
    // 存储到 IndexedDB
    const db = await this.getDB()
    await db.changes.add({
      ...changes,
      timestamp: Date.now(),
      synced: false,
    })
  }

  /**
   * 获取待同步变更
   */
  private async getPendingChanges(): Promise<any[]> {
    const db = await this.getDB()
    return db.changes.where('synced').equals(false).toArray()
  }
```
